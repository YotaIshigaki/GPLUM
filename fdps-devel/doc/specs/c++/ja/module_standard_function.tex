\subsubsubsection{時間計測}

\subsubsubsubsection{PS::GetWtime}

\begin{screen}
\begin{verbatim}
inline PS::F64 PS::GetWtime();
\end{verbatim}
\end{screen}

\begin{itemize}

\item{{\bf 引数}}

なし。

\item{{\bf 返り値}}

{PS::F64}型。ウォールクロックタイムを返す。単位は秒。

\end{itemize}
