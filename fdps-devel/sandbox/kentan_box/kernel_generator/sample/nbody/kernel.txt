EPI F32vec xi:pos
EPI F32    eps2i:eps2
EPJ F32vec xj:pos
EPJ F32    mj:mass
EPJ F32    eps2j:eps2
FORCE F32vec f:acc
FORCE F32 phi:pot

function sub(a,b)
  return a-b
end

rij = xi - xj
r2 = eps2i + eps2j + rij*rij
#pragma statement fission_point
rinv = rsqrt(r2)
#pragma statement fission_point
mrinv = mj*rinv
f -= mrinv*rinv*rinv * rij
phi -= mrinv
